// 量化推理函数
void quantized_inference(
    const int8_t* weight, const int8_t* input, const float* bias,
    float scale_w, float scale_x, int zero_point_w, int zero_point_x,
    int weight_rows, int weight_cols, float* output)
{
    // 中间结果累积使用 int32_t
    int32_t y_int32[weight_rows];

    // 初始化输出为 0
    for (int i = 0; i < weight_rows; i++) {
        y_int32[i] = 0;
    }

    // 矩阵乘法：计算量化的矩阵乘法结果
    for (int i = 0; i < weight_rows; i++) {
        for (int j = 0; j < weight_cols; j++) {
            // 累积乘法：输出 = 量化权重 * 量化输入
            y_int32[i] += (weight[i * weight_cols + j] - zero_point_w) *
                          (input[j] - zero_point_x);
        }
    }

    // 反量化：将 int32 转换为浮点数，并加上偏置
    for (int i = 0; i < weight_rows; i++) {
        output[i] = scale_w * scale_x * y_int32[i] + bias[i];
    }
}
